# Function to find the closest match in 'loc_df' for each hospital
find_closest_match <- function(hospital_name, reference_df, name_column) {
  # Calculate the string distance between 'hospital_name' and all names in 'reference_df'
  distances <- stringdist::stringdist(hospital_name, reference_df[[name_column]], method = "jw")
  
  # Find the index of the minimum distance
  closest_index <- which.min(distances)
  
  # Return the closest match
  closest_match <- reference_df[closest_index, ]
  return(closest_match)
}

# Add new columns based on the closest match
hospital_loc <- hospital_loc %>%
  rowwise() %>%  # Important for applying the function to each row independently
  mutate(
    closest_match = list(find_closest_match(hosp, loc_df, "ENGLISH_NA")),
    hospital_loc_x = closest_match$X,
    hospital_loc_y = closest_match$Y,
    hospital_reference = closest_match$ENGLISH_NA
  ) %>%
  select(-closest_match)  # Remove the list column

# Initialize an empty dataframe to store the results
h_to_h_dist <- data.frame(
  referring = character(),
  accepting = character(),
  distance = numeric(),
  stringsAsFactors = FALSE
)

# Calculate the distances between all pairs of referring and accepting hospitals
for (i in 1:nrow(referring_hospitals)) {
  for (j in 1:nrow(accepting_hospitals)) {
    dist <- distHaversine(
      c(referring_hospitals$hospital_loc_x[i], referring_hospitals$hospital_loc_y[i]),
      c(accepting_hospitals$hospital_loc_x[j], accepting_hospitals$hospital_loc_y[j])
    )/1000
    h_to_h_dist <- rbind(h_to_h_dist, data.frame(
      referring = referring_hospitals$hospital[i],
      accepting = accepting_hospitals$hospital[j],
      distance = dist,
      stringsAsFactors = FALSE
    ))
  }
}

# Load the data
data3 <- read.csv("lhin_income_data.csv", header = TRUE)

# Initialize an empty vector to store median incomes
median_incomes <- numeric(nrow(data3))

# Loop through each row (LHIN region)
for (i in 1:nrow(data3)) {
  # Extract income brackets and frequencies for the region
  income_brackets <- names(data3)[3:ncol(data3)]
  frequencies <- as.numeric(data3[i, 3:ncol(data3)])
  
  # Expand the income brackets according to their frequencies
  expanded_incomes <- rep(income_brackets, frequencies)
  
  # Convert income brackets to numeric values
  # Note: Replace non-numeric characters and convert to numeric
  expanded_incomes_numeric <- as.numeric(gsub("[^0-9]", "", expanded_incomes))
  
  # Calculate the median income for the region
  median_incomes[i] <- median(expanded_incomes_numeric)
}

# Add the median incomes to the data frame
data3$Median_Income <- median_incomes

# Print the updated data frame
#print(data3)

write.csv(data3, file = "lhin_data_update.csv", row.names = FALSE)
